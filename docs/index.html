Integrated in Minutes

quickstart.py
from engram.events import MemoryEventManager
from engram.schemas import MemoryEntryCreate
from engram import AsyncSessionLocal

# Write a structured decision into the durable memory ledger
async with AsyncSessionLocal() as session:
    mgr = MemoryEventManager(session)
    entry = await mgr.add_memory_entry(
        MemoryEntryCreate(
            type="decision",
            title="Use SQLAlchemy Async",
            content="Enables parallel database transactional writes on the AWS ops instance.",
            tags=["infrastructure", "orm", "db"],
            references=["task-245-caddy"],
            source_file="ops/caddy/config.py",
        )
    )

# Query memories dynamically inside retrieval hooks
from engram.hooks import MemoryRetrievalHooks

async with AsyncSessionLocal() as session:
    hooks = MemoryRetrievalHooks(session)
    
    # Inject on session-start
    ctx = await hooks.session_start_hook(limit=10)
    
    # Inline semantic tag matching on incoming user messages
    message_ctx = await hooks.per_message_hook(tags=["infrastructure"], limit=5)
